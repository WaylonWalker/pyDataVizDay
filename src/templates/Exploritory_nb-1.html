## define dollar_formatter

from matplotlib.ticker import FuncFormatter
import math

def millify(n, pos=1):
    """
    Function to shorten long numbers as strings with alphabetical scale designator.  This function was taken from a
    stack overflow post a long time ago.  I have since modified it slightly but is still fairly true to to original.

    :param n: number to millify
    :param pos: Ignore the passed in position. This has the effect of scaling the default tick locations.
    :return:

    Examples:

    .. ipython:: python

        import pea
        import pandas as pd
        print(pea.millify(1234567))
        print(pea.millify(1234567890))
        bigdf = pd.DataFrame({'Big Numbers':[123, 123456, 123456789, 1234567890]})
        bigdf['millified'] = bigdf['Big Numbers'].apply(lambda x: pea.millify(x))
        bigdf

    """

    n = float(n)

    mill_names = ['  ', ' K', ' M', ' B', ' T']

    millidx = max(0, min(len(mill_names) - 1,
                         float(math.floor(0 if n == 0 else math.log10(abs(n)) / 3))))

    milled = '{:.2f}{}'.format(n / 10 ** (3 * millidx), mill_names[int(millidx)])

    if milled[-3] == '0':
        milled = '{:.1f}{}'.format(n / 10 ** (3 * millidx), mill_names[int(millidx)]) + ' '
    if milled[-4] == '0':
        milled = '{:.0f}{}'.format(n / 10 ** (3 * millidx), mill_names[int(millidx)]) + ' '

    return milled

def to_dollars(y, position=1):
    s = millify(y)
    return '$' + s

dollar_formatter = FuncFormatter(to_dollars)
